pipeline {
    agent any
    tools {
        jdk 'system-jdk21'
    }
    environment {
        BACK_HOST = "172.20.3.10"
        BACK_USER = "fitness"
        BACK_PATH = "/opt/backend-dev/app.jar"
        CREDENTIALS_ID = "ssh-back-dev"
    }
    stages {
        stage('Clone') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/FitGR/fit-back.git'
            }
        }
        stage('Build') {
            steps {
                sh 'chmod +x mvnw'
                sh './mvnw clean package -DskipTests'
            }
        }
        stage('Deploy') {
            when {
                expression { fileExists('target/demojenkins-0.0.1-SNAPSHOT.jar') }
            }
            steps {
                sshagent(credentials: [env.CREDENTIALS_ID]) {
                    sh """
                        echo " Copying JAR to backend VM..."
                        scp -o StrictHostKeyChecking=no target/demojenkins-0.0.1-SNAPSHOT.jar \
                            ${BACK_USER}@${BACK_HOST}:${BACK_PATH}
                        echo " Restarting backend service..."
                        ssh -o StrictHostKeyChecking=no ${BACK_USER}@${BACK_HOST} \
                            "sudo systemctl restart backend-dev"
                    """
                }
            }
        }
    }
    post {
        success {
            echo "deployment succeeded!"
        }
        failure {
            echo "deployment failed!"
        }
    }
}
